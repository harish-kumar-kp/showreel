global proc matte()
{
    // Make a new window
    if ( `window -exists MyWindow` ) {
     deleteUI MyWindow;
    }
    window -t "MatteEnable/Disable" MyWindow;
       //-iconName "Short Name"
       //-widthHeight 250 210;
    columnLayout -adjustableColumn true;
       text("Selected Objects");
       text(".-.");
       button -label "Enable Matte" -command Enable;
       button -label "Disable Matte" -command Disable;
     
       text(".....");
       text("All Shaders");
       text("_");
       button -label "Enable Matte" -command AllEnable;
       button -label "Disable Matte" -command AllDisable;
       text(".");
       text("___");
       button -label "Exit" -command ("deleteUI -window MyWindow");
    setParent ..;
    showWindow MyWindow;
    // Resize the main window
    //
    //global string $gMainWindow;
    //window -edit -widthHeight 900 777 $gMainWindow;
  
   
}
global proc Enable( ) {
   
    string $materials[] = stringArrayRemoveDuplicates(ls("-mat",listConnections(listConnections("-type","shadingEngine",`ls -sl -o -dag -s`))));
    if(size($materials)==0){
        print("No Object is Selected to Enable");
       
    }
    for($material in $materials){
        editRenderLayerAdjustment ($material+".aiEnableMatte");
        setAttr ($material+".aiEnableMatte") 1;
        print("Matte For Selected is Enabled");
    }

}
global proc Disable( ) {
   
    string $materials[] = stringArrayRemoveDuplicates(ls("-mat",listConnections(listConnections("-type","shadingEngine",`ls -sl -o -dag -s`))));
       if(size($materials)==0){
        print("No Object is Selected to Disable");
       
    }
    for($material in $materials){
        editRenderLayerAdjustment -remove($material+".aiEnableMatte");
        setAttr ($material+".aiEnableMatte") 0;
        print("Matte For Selected is Disabled");
    }

}
global proc AllEnable( ) {
    //Get ShaderSG Name List
    string $shaderSG[] = `ls -type shadingEngine`;
    for($SG in $shaderSG){
        string $shaders[] = `listConnections ($SG+".surfaceShader")`;
        for($shader in $shaders){
            if($shader != "lambert1" && $shader != "RedMat_Shader" && $shader != "GreenMat_Shader" && $shader != "BlueMat_Shader" && $shader != "CyanMat_Shader" && $shader != "MagentaMat_Shader" && $shader != "WhiteMat_Shader" && $shader != "YellowMat_Shader" && $shader != "CutMat_Shader" && $shader != "ShadowMask_Shader" ){
                editRenderLayerAdjustment ($shader+".aiEnableMatte");
                setAttr ($shader+".aiEnableMatte") 1;
            }
        }
    }
   
print("Matte For All Shaders Enabled");
}
global proc AllDisable( ) {
    //Get ShaderSG Name List
    string $shaderSG[] = `ls -type shadingEngine`;
    for($SG in $shaderSG){
        string $shaders[] = `listConnections ($SG+".surfaceShader")`;
        for($shader in $shaders){
            if($shader != "lambert1" && $shader != "RedMat_Shader" && $shader != "GreenMat_Shader" && $shader != "BlueMat_Shader" && $shader != "CyanMat_Shader" && $shader != "MagentaMat_Shader" && $shader != "WhiteMat_Shader" && $shader != "YellowMat_Shader" && $shader != "CutMat_Shader" && $shader != "ShadowMask_Shader" ){
                editRenderLayerAdjustment -remove ($shader+".aiEnableMatte");
                setAttr ($shader+".aiEnableMatte") 0;
            }
        }
    }
   
print("Matte For All Shaders Disabled");
}


MEL Script:"Catche Exporter"

global proc createCache()
{
string $objSel[] = `ls -sl`;
string $objName=`textField -q -text objName`;
string $pathName=`textField -q -text pathName`;
 
performCreateGeometryCache 1 "add";
cacheSetDirectory  ($pathName) "directory";
hideOptionBox;
 
select -r $objSel[0];
select -hi;
doCreateGeometryCache 4 { "2", "1", "10", "OneFile", "1", ($pathName + "/" +$objName),"0",($objName + "_geoCache"),"0", "add", "0", "1", "1","0","1" } ;
 
select -r $objSel[0];
print "Geometry Cache file Exported";
};
 
//--------------Baking Mesh Transform deformation and Exporting the keys---------------
 
global proc bakeKeys()
{
string $objSel[] = `ls -sl`;
string $objNameA=`textField -q -text objNameA`;
string $pathNameA=`textField -q -text pathNameA`;
 
bakeResults -simulation true -t "1:25" -hierarchy below -sampleBy 1 -disableImplicitControl true -preserveOutsideKeys true -sparseAnimCurveBake false -controlPoints false -shape true {$objSel[0]};
 
select -r $objSel[0];
file -op "precision=8;intValue=17;nodeNames=1;verboseUnits=0;whichRange=1;range=0:10;options=keys;hierarchy=below;controlPoints=0;shapes=1;helpPictures=0;useChannelBox=0;copyKeyCmd=-animation objects -option keys -hierarchy below -controlPoints 0 -shape 1 " -typ "animExport" -pr -es ($pathNameA + "/" +$objNameA + "/" + $objNameA);
 
select -r $objSel[0];
print "Anim file exported";
 
};
